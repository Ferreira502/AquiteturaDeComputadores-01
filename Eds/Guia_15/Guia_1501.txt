// Operação: dado04 = dado01 - dado02 - dado03
// Dados de 8 bits

        LXI   H, dado01   ; HL <- endereço de dado01
        MOV   A, M        ; A <- MEM[dado01]

        INX   H           ; HL <- endereço de dado02
        SUB   M           ; A <- A - MEM[dado02]

        INX   H           ; HL <- endereço de dado03
        SUB   M           ; A <- A - MEM[dado03]

        STA   dado04      ; MEM[dado04] <- A   (salva o resultado)

        HLT               ; fim do programa


; Area de dados
dado01:  05h    ; exemplo: 5
dado02:  02h    ; exemplo: 2
dado03:  01h    ; exemplo: 1
dado04:  00h    ; (recebe o resultado)
